import { cn } from "@/lib/utils"
import { Button } from "@/components/ui/button"
import { LogoutButton } from "@/components/ui/logout-button"
import {
  BarChart,
  Users,
  Soup,
  LeafyGreen,
} from "lucide-react"

interface SidebarProps extends React.HTMLAttributes<HTMLDivElement> {
  activeView: string
  setActiveView: (view: string) => void
}

/**
 * A sidebar component for the manager view.
 *
 * This component displays a list of buttons for each of the manager views. The
 * button for the currently active view is highlighted. The component also
 * displays a logout button at the bottom.
 *
 * @param className - The CSS class to apply to the root element of the component.
 * @param activeView - The currently active view.
 * @param setActiveView - A function to set the active view.
 */
export function Sidebar({ className, activeView, setActiveView }: SidebarProps) {
  return (
    <div className={cn("pb-12 w-64 bg-white", className)}>
      <div className="space-y-4 py-4">
        <div className="px-3 py-2">
          <h2 className="mb-2 px-4 text-lg font-semibold tracking-tight">
            Manager Dashboard
          </h2>
          <div className="space-y-1">
            <Button
              variant={activeView === "Reports" ? "default" : "ghost"}
              className="w-full justify-start py-6"
              onClick={() => setActiveView("Reports")}
            >
              <BarChart className="mr-2 h-4 w-4" />
              Reports
            </Button>
            <Button
              variant={activeView === "Employees" ? "default" : "ghost"}
              className="w-full justify-start py-6"
              onClick={() => setActiveView("Employees")}
            >
              <Users className="mr-2 h-4 w-4" />
              Employees
            </Button>
            <Button
              variant={activeView === "Menu Items" ? "default" : "ghost"}
              className="w-full justify-start py-6"
              onClick={() => setActiveView("Menu Items")}
            >
              <Soup className="mr-2 h-4 w-4" />
              Menu Items
            </Button>
            <Button
              variant={activeView === "Ingredients" ? "default" : "ghost"}
              className="w-full justify-start py-6"
              onClick={() => setActiveView("Ingredients")}
            >
              <LeafyGreen className="mr-2 h-4 w-4" />
              Ingredients
            </Button>
          </div>
        </div>
      </div>
      <div className="px-3 py-2">
        <LogoutButton className="w-full justify-start" />
      </div>
    </div>
  )
}
