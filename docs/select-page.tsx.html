'use client'

import { useRouter } from 'next/navigation'
import { Button } from "@/components/ui/button"
import { LogoutButton } from "@/components/ui/logout-button"
import { ChartBar, Tablet, ChefHat, SquareMenu, CircleDollarSign } from 'lucide-react'

const views = [
  { name: 'Manager', path: '/manager', icon: ChartBar },
  { name: 'Kiosk', path: '/kiosk', icon: Tablet },
  { name: 'Kitchen Display', path: '/kitchen-display', icon: ChefHat },
  { name: 'Menu', path: '/menu', icon: SquareMenu },
  { name: 'Cashier', path: '/cashier', icon: CircleDollarSign },
]

/**
 * Page allowing the user to select the device view (manager, kiosk, kitchen display, menu, cashier).
 * 
 * @returns {JSX.Element} The JSX element representing the SelectPage component.
 */
export function SelectPage() {
  const router = useRouter()

  return (
    <div className="min-h-screen bg-background flex flex-col items-center justify-center p-4">
      <div className="absolute top-4 right-4">
        <LogoutButton />
      </div>
      <h1 className="text-3xl font-bold mb-8">Select Device View</h1>
      <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-5 gap-6">
        {views.map((view) => (
          <Button
            key={view.name}
            variant="outline"
            className="h-32 w-32 flex flex-col items-center justify-center text-center p-2"
            onClick={() => router.push(view.path)}
          >
            <view.icon className="h-12 w-12 mb-2" />
            <span>{view.name}</span>
          </Button>
        ))}
      </div>
    </div>
  )
}
