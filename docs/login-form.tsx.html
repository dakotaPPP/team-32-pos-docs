'use client'

import { Button } from "@/components/ui/button"
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card"
import { signIn } from "next-auth/react"
import Image from "next/image"

/**
 * A login form component which renders a centered card with a Google sign in button.
 * 
 * This component uses the {@link https://next-auth.js.org/ `next-auth`} library to handle the authentication flow.
 * 
 * @returns A JSX element representing the login form.
 */
export function LoginForm() {

  /**
   * Handles the Google login process.
   * 
   * This function uses the {@link https://next-auth.js.org/ `next-auth`} library to handle the authentication flow.
   * When the user clicks the Google sign in button, this function is called and it redirects the user to the Google
   * authentication page. Once the user has authenticated, they are redirected back to this page with an authorization
   * token which is then used to authenticate the user with the server.
   * 
   * If there is an error during the authentication flow, the error is logged to the console.
   */
  const handleGoogleLogin = async () => {
    try {
      await signIn('google', { 
        callbackUrl: '/select',
        redirect: true,
      })
    } catch (error) {
      console.error('Login error:', error)
    }
  }

  return (
    <div className="flex items-center justify-center min-h-screen bg-background">
      <Card className="w-[350px]">
        <CardHeader className="text-center">
          <CardTitle>Welcome to Team 32(1) POS</CardTitle>
          <CardDescription>Sign in to access your account</CardDescription>
        </CardHeader>
        <CardContent>
          <div className="flex justify-center mb-4">
            <Image src="https://i.imgur.com/fYf1GtG.png" alt="Panda Express Logo" width={100} height={100}/>
          </div>
          <Button 
            className="w-full" 
            onClick={handleGoogleLogin}
          >
            <svg className="mr-2 h-4 w-4" aria-hidden="true" focusable="false" data-prefix="fab" data-icon="google" role="img" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 488 512">
              <path fill="currentColor" d="M488 261.8C488 403.3 391.1 504 248 504 110.8 504 0 393.2 0 256S110.8 8 248 8c66.8 0 123 24.5 166.3 64.9l-67.5 64.9C258.5 52.6 94.3 116.6 94.3 256c0 86.5 69.1 156.6 153.7 156.6 98.2 0 135-70.4 140.8-106.9H248v-85.3h236.1c2.3 12.7 3.9 24.9 3.9 41.4z"></path>
            </svg>
            Sign in with Google
          </Button>
        </CardContent>
      </Card>
    </div>
  )
}
